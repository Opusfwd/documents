import * as _ from 'lodash';
import {
  experience,
  condition,
  backPain,
  knee,
} from '@/screens/application/tablet/tabs/yourDetail/healthquestion/DropDownInfo';

const commonOptions = ['Yes', 'No'];

const optionMap = {
  OPTION1: experience,
  OPTION2: condition,
  OPTION3: backPain,
  OPTION4: knee,
};

const getQuestionTitle = ({ questionType, isOwner }) => {
  const type = /FULL/.test(questionType) ? 'FUW' : 'SIO';
  const relationship = isOwner ? 'CO' : 'PC';
  return `Health Question_${type}_${relationship}`;
};

const getHealthQuestion = ({ isPayer, isOwner, isInsured, HQInfo, applicationData }) => {
  if ((!isOwner && isPayer) || !HQInfo) {
    return [];
  }

  let healthQuestion, questionPart;

  // PC || CO === PC
  if (isInsured || applicationData?.insureds?.[0]?.relationship === 'SELF') {
    // PC
    healthQuestion = HQInfo?.proposersHQ;
    questionPart = HQInfo?.questionPartHQ;
  } else {
    // CO
    healthQuestion = HQInfo?.childHQ;
    questionPart = HQInfo?.questionPartForOwnerHQ;
  }

  const name = getQuestionTitle({
    questionType: HQInfo?.questionType,
    isOwner,
  });

  if (!healthQuestion) {
    return [];
  }

  const { questions, enquiryLines } = healthQuestion?.reduce(
    (data, item, index) => {
      if (item.isShow) {
        const { questions, enquiryLines } = data;
        const itemSection = getQuestion({ item, name });
        const partSection = getQuestionPart({ index, item, name, questionPart });
        const [ques, enquirys] = flattenQuestion(
          _.compact([...partSection, itemSection]),
          name,
          questions,
          enquiryLines,
        );
        return {
          questions: ques,
          enquiryLines: enquirys,
        };
      }
      return data;
    },
    {
      questions: [],
      enquiryLines: [],
    },
  );

  return !questions?.length && !enquiryLines.length
    ? []
    : [
        {
          name: name,
          isSatisfied: true,
          hasQuestions: true,
          enquiryLines: [
            {
              name: name,
              path: name,
              section: name,
              isSatisfied: true,
              hasQuestions: true,
              isRoot: true,
              triggers: [],
              triggerQuestions: {},
              questions,
            },
            ...enquiryLines,
          ],
        },
      ];
};

const getQuestion = ({ item, name }) => {
  const itemSection = {
    ..._.pick(item, ['name', 'path', 'locale', 'isSatisfied', 'answers', 'definition']),
    hasAnswer: !!item.answers,
    triggers: [name],
  };
  if (item.isCheckRuleAnswer && item.ruleAnswer?.length) {
    itemSection.questions = item.ruleAnswer?.reduce((data, el) => {
      if (el.isShow) {
        const name = el.text;
        const answers = _.compact([el.answer, el.answer2]);
        const newAnswers = optionMap[el.type]
          ? answers?.map((ee) => optionMap[el.type].find((dl) => dl.name === ee)?.labelKey)
          : answers;
        const type = getType(el.type);
        const options = getOptions(el?.type, el.options, newAnswers);
        return [
          ...data,
          {
            name: name,
            path: name,
            section: name,
            triggers: [itemSection.name],
            hasAnswer: true,
            triggerQuestions: {},
            isSatisfied: !!el?.isSatisfied,
            answers: newAnswers,
            definition: {
              name: el.name,
              text: el.text,
              type: type,
              isMultiValued: answers?.length > 1 ? true : false,
              options: options,
            },
          },
        ];
      }
      return data;
    }, []);
  }
  if (item.second?.isShow && item.second?.sections?.length) {
    itemSection.questions = item.second?.sections?.reduce((data, el) => {
      if (el.isShow) {
        let questions = [];
        const name = el.title || el.placeholder;
        if (el?.answer?.length && el?.answer?.[0]?.text) {
          questions = el?.answer?.reduce((ans, dd) => {
            const answer = getAnswer(dd.type, dd.answer);
            const options = getOptions(dd?.type, dd.options, [answer]);
            if (dd.isShow) {
              return [
                ...ans,
                {
                  name: dd.text,
                  path: dd.text,
                  section: dd.text,
                  triggers: [name],
                  hasAnswer: !!answer,
                  triggerQuestions: {},
                  answers: [answer],
                  definition: {
                    name: dd.text,
                    text: dd.text,
                    type: getType(dd.type),
                    options,
                  },
                },
              ];
            }
            return ans;
          }, []);
        }
        if (el?.childSections?.length && el?.childSections?.[0]?.title) {
          questions = el?.childSections?.reduce((childs, dd) => {
            if (dd.isShow) {
              return [
                ...childs,
                {
                  name: dd.title,
                  path: dd.title,
                  section: dd.title,
                  triggers: [name],
                  hasAnswer: !!dd.answer,
                  triggerQuestions: {},
                  answers: [dd.answer],
                  definition: {
                    name: dd.title,
                    text: dd.title,
                    type: getType(dd.childType),
                    options: dd.options,
                  },
                },
              ];
            }
            return childs;
          }, []);
        }
        return [
          ...data,
          {
            name: name,
            path: name,
            section: name,
            triggers: [itemSection.name],
            hasAnswer: true,
            triggerQuestions: {},
            answers: el.type === 'CHECKBOX' ? [commonOptions[el.isCheck ? 0 : 1]] : [el.answer],
            definition: {
              name: name,
              text: name,
              type: getType(el.type),
              options: el.type === 'CHECKBOX' ? commonOptions : el.options,
            },
            questions,
          },
        ];
      }
      return data;
    }, []);
  }

  return itemSection;
};

const getQuestionPart = ({ index, name, item, questionPart }) => {
  const partSection = [];
  if (index === 3 && questionPart) {
    const partName = 'Have you ever been diagnosed with or had symptoms of any of the following';
    const hasAnswer = !!questionPart?.find((el) => el.answer && el.answer === 'Yes');
    const partSectionTemp = {
      name: partName,
      path: partName,
      section: partName,
      triggers: [name],
      hasAnswer: true,
      triggerQuestions: {},
      answers: [commonOptions[hasAnswer ? 0 : 1]],
      definition: {
        name: partName,
        text: partName,
        options: commonOptions,
      },
      questions: questionPart.map((sec) => ({
        name: sec.name,
        path: sec.name,
        locale: item.locale,
        hasAnswer: !!sec?.answer,
        isSatisfied: !!sec?.isSatisfied,
        answers: [sec?.answer],
        triggers: [partName],
        definition: {
          name: sec?.name,
          text: sec?.text,
          isMultiValued: false,
          options: commonOptions,
        },
        questions: sec?.second?.reduce((ques, second) => {
          if (second.isShow) {
            const answers = commonOptions[second?.isCheck ? 0 : 1];
            return [
              ...ques,
              {
                name: second.name,
                path: second.name,
                locale: item.locale,
                hasAnswer: !!answers,
                isSatisfied: !!second?.isSatisfied,
                answers: [answers],
                triggers: [sec?.name],
                definition: {
                  name: second.name,
                  text: second.name,
                  isMultiValued: false,
                  options: commonOptions,
                },
                questions: second?.answer?.reduce((ans, el) => {
                  if (el.isShow) {
                    const answers = _.compact([el.answer, el.answer2]);
                    const newAnswers = optionMap[el.type]
                      ? answers?.map((ee) => optionMap[el.type].find((dl) => dl.name === ee)?.labelKey)
                      : answers;
                    const type = getType(el.type);
                    const options = getOptions(el?.type, el.options, newAnswers);
                    return [
                      ...ans,
                      {
                        name: el.name,
                        path: el.name,
                        locale: item.locale,
                        hasAnswer: !!newAnswers,
                        isSatisfied: !!el?.isSatisfied,
                        answers: newAnswers,
                        triggers: [second.name],
                        definition: {
                          name: el.name,
                          text: el.text,
                          type: type,
                          isMultiValued: answers?.length > 1 ? true : false,
                          options: options,
                        },
                      },
                    ];
                  }
                  return ans;
                }, []),
              },
            ];
          }
          return ques;
        }, []),
      })),
    };
    partSection.push(partSectionTemp);
  }
  return partSection;
};

const getType = (type) => {
  if (optionMap[type] || type === 'CHECKBOX') {
    return 'OPTION_GROUP';
  }
  if (type === 'STRINGGROUP') {
    return 'STRING_GROUP';
  }
  if (type === 'STRING') {
    return 'NUMBER';
  }
  return type;
};

const getAnswer = (type, answer) => {
  const optionList = optionMap[type];
  const newAnswer = optionList ? optionList.find((el) => el.name === answer)?.labelKey : answer;
  return newAnswer;
};

const getOptions = (type, options, answers) => {
  const newOptions = optionMap[type]?.map?.(({ labelKey }) => labelKey) || options;
  if (type === 'OPTION_GROUP') {
    return commonOptions;
  }
  if (type === 'STRINGGROUP' && answers) {
    return answers;
  }
  return newOptions;
};

const flattenQuestion = (questionsArrs, rootName, questions, enquiryLines) => {
  const flatSection = (item, rootItem) => {
    const rootData = rootItem || item;
    const parentName = item.triggers?.[0];
    const isRoot = item.triggers?.[0] === rootName;
    const data = _.omit(item, 'questions');
    if (data.definition?.options?.includes('No')) {
      data.definition.options = data.definition?.options.sort((a, b) => b.localeCompare(a));
      if (!data.definition?.type) {
        data.definition.type = 'OPTION_GROUP';
      }
    }

    if (isRoot) {
      questions.push(data);
    } else {
      const parentIndex = enquiryLines?.findIndex((el) => el.name === parentName);
      if (parentIndex > -1) {
        enquiryLines[parentIndex].questions.push(data);
      } else {
        enquiryLines.push({
          name: parentName,
          path: parentName,
          section: parentName,
          triggers: [parentName],
          isSatisfied: true,
          hasQuestions: true,
          isRoot: false,
          triggerQuestions: {},
          referAnswer: rootData?.answers?.[0],
          questions: [data],
        });
      }
    }
    if (item.questions) {
      item.questions?.map((el) => flatSection(el, item));
    }
  };
  questionsArrs?.map((item) => flatSection(item));
  return [questions, enquiryLines];
};

export default getHealthQuestion;
